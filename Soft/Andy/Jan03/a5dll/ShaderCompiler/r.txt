; res.w holds squared source length
func Normalize3 res
	locals rLeng
	dp3 res.w, res, res
	rsq rLeng.w, res.w
	mul res.xyz, res.xyz, rLeng.w

; res.w holds squared source length, leng.w holds length multiplier
func Normalize3Mul res leng
	locals rLeng
	dp3 res.w, res, res
	rsq rLeng.w, res.w
	mul rLeng.w, rLeng.w, leng.w
	mul res.xyz, res.xyz, rLeng.w

// res.w has squared distance between s1 & s2
func AddNormalize3 res s1 s2
	locals rLeng
	add res.xyz, s1.xyz, s2.xyz
	Normalize3 res

func CalcCameraDir res
	AddNormalize3 res, c6, -v0

// result is stored to res.z
func CalcSpecular res LightDir HalfVector SpecParams
	dp3 res.x, v1, LightDir
	dp3 res.y, v1, HalfVector
	mov res.w, SpecParams.w
	mad res.y, res.y, SpecParams.y, SpecParams.x // r3.y = a + b * (N*H) a = c16.x  b = c16.y
	lit res, res
	mul res.z, res.z, SpecParams.z

proc DirSpecularLight
	m4x4 oPos, v0, c10
	locals cameraDir, halfVector, litRes
	CalcCameraDir cameraDir
	AddNormalize3 halfVector cameraDir c15
	CalcSpecular litRes, c15, halfVector, c16
	mul litRes, litRes.z, c17
	mul oD0, litRes, v2

proc DirSpecularLightTex

	mov oT0, v3

proc PointSpecularLight
	m4x4 oPos, v0, c10
	locals cameraDir, lightDir, halfVector, litRes
	CalcCameraDir cameraDir
	AddNormalize3 lightDir, c14, -v0
	AddNormalize3 halfVector, cameraDir, lightDir
	CalcSpecular litRes, lightDir, halfVector, c16
; account distance fade
	mul lightDir.w, lightDir.w, c17.w // r3.w = sqr(dist/R)
	rcp lightDir.w, lightDir.w
	mul lightDir.w, lightDir.w, c4.y        // r3.w = sqr(R/leng/4)
	mul litRes.z, litRes.z, lightDir.w
	mul oD0, litRes.z, c15

proc NHCalcer
	m4x4 oPos, v0, c10
	locals cameraDir, lightDir, halfVector

	mad lightDir.xyz, -v0.xyz, c14.www, c14.xyz // r5 = (L),  world space

	mov oT0.xyz, v1  // N
	mov oT1.xyz, halfVector  // H
	mad oT2.xyz, lightDir, c2.xxx, c2.xxx  // L

func CalcLocalVector res globalVec

proc FastPointBumpLight
	m4x4 oPos, v0, c10
	locals lightDir, localNormal
	sub lightDir.xyz, c14.xyz, v0.xyz
	dp3 lightDir.w, lightDir, lightDir
	mul lightDir.w, lightDir.w, c15.w       // sqr(leng/R)
	rcp lightDir.w, lightDir.w
	mul lightDir.w, lightDir.w, c4.y        // sqr(R/leng/4)
	max lightDir.w, lightDir.w, c0.w         // clamp attenuation, lightDir.w holds it
; second part
	CalcLocalVector localNormal lightDir
	Normalize3Mul localNormal, lightDir
	mad oD0.xyz, localNormal.xyz, c2.xxx, c2.xxx
	mov oT0.xy, v3.xy
	mul oD1.xyz, c16, v2

; params.x = h - distance to plane
; att.x = B = R^2 / (16*h^2)
; att.y = alpha = B > 1 ? 1/sqrt(B) : 1
func CalcPointLightMapping lightDir, params, att
	sub lightDir.xyz, c14, v0       // r0 = dir to light center
	dp3 params.x, lightDir, v1      // params.x = h(distance to plane)
	mul params.w, params.x, params.x
	rcp params.w, params.w
	mul params.w, params.w, c15.y
	mul params.w, params.w, c4.y    // params.w = B
	sge params.z, params.w, c1.w    // params.z = params.w > 1
	add params.y, -params.z, c1.w   // params.y = params.w < 1
	mad att.x, params.y, params.w, params.z // final B
	// calc alpha
	rsq att.y, params.w
	mad att.y, params.z, att.y, params.y // final alpha

func CalcPointLightTex lightDir, params, att
	locals texCoord
	dp3 texCoord.x, v4, lightDir
	dp3 texCoord.y, v5, lightDir
	max texCoord.w, params.x, -params.x
	rcp texCoord.w, texCoord.w
	mul texCoord.w, texCoord.w, att.y
	mul texCoord.w, texCoord.w, c4.x
	mul texCoord.xy, texCoord.xy, texCoord.w
	mad oT0.xy, texCoord.xyz, c2.xx, c2.xx 

; B = R^2 / (16*h^2),  h - distance to plane, x - distance in plane
; result = B / (1+x^2)  is replaced with 1 / (1+(alpha*x)^2)
; alpha = B > 1 ? 1/sqrt(B) : 1
proc PointFalloff
	m4x4 oPos, v0, c10
	locals lightDir, temp, att, texCoord
	CalcPointLightMapping lightDir, temp, att
	mul oD0, att.x, c16    // attenutation to diffuse
	CalcPointLightTex lightDir, temp, att


; B = R^2 / (16*h^2),  h - distance to plane, x - distance in plane
; result = B / ((1+x^2)^(3/2))  is replaced with 1 / ((1+(alpha*x)^2)^(3/2))
; alpha = 1 / sqrt(B) if B > 1, alpha = 1 otherwise
; same approximation is used in both cases (for falloff and light)
proc PointLight
	m4x4 oPos, v0, c10
	locals lightDir, temp, att, color
	CalcPointLightMapping lightDir, temp, att
	sge att.z, temp.x, c0.x       // att.z = h > 0
	mul att.z, att.x, att.z       // att.z = final B capped
	mul color.xyz, att.z, c16
	mul oD0.xyz, color, v2        // attenutation to diffuse
	CalcPointLightTex lightDir, temp, att

func DoMirror outTex
	locals cameraDir, vFresnel
	CalcCameraDir cameraDir
	// r1 = cameraDir
	dp3 cameraDir.w, cameraDir, v1
	add vFresnel.w, c1.x, -cameraDir.w
	mul vFresnel.w, vFresnel.w, vFresnel.w
	mul vFresnel.w, vFresnel.w, vFresnel.w
	mul cameraDir.w, cameraDir.w, c2.z // cameraDir.w = 2 * ( cameraDir dp3 v1 )
	mad oD0.w, vFresnel.w, c14.z, c14.w
	mad outTex.xyz, cameraDir.w, v1, -cameraDir.xyz

[PS] CalcNH
ps.1.1
tex t0 // N
tex t1 // H
texcoord t2 // L
sub r1.rgb, t0, t1  // r1 = a/2; a = N-H
dp3 r1, r1, r1  // (a/2)*(a/2)
sub_x2 r0.rgb, t0, t1
+mad r1.a, c0, r1, c0 // c0.a = 0.5
dp3 r0, r0, r0  // (a)*(a)
mov r0.rgb, c0           // ro is stored to rgb, c0 has ro, or t0.a for bump
+cnd r0.a, r0.a, r1, r0  // alpha channel is subject, f(NH) is stored to alpha
dp3_x4 r1, t0_bx2, t2_bx2
mov_x4_sat r1, r1
lrp r0.a, r1.a, r0, c1
[RS]
D3DRS_ALPHATESTENABLE FALSE

[PS] CalcNHBumped
