func CalcDynamicAmbient outRes normal
	locals res, dir, dir1
	max dir.xyz, normal, c0
	mul res.xyz, c25, dir.x
	mad res.xyz, c27, dir.y, res
	mad res.xyz, c29, dir.z, res
	max dir1.xyz, -normal, c0
	mad res.xyz, c26, dir1.x, res
	mad res.xyz, c28, dir1.y, res
	mad outRes.xyz, c30, dir1.z, res

; res.w holds squared source length
func Normalize3 res
	locals rLeng
	dp3 res.w, res, res
	rsq rLeng.w, res.w
	mul res.xyz, res.xyz, rLeng.w

; res.w holds squared source length, leng.w holds length multiplier
func Normalize3Mul res leng
	locals rLeng
	dp3 res.w, res, res
	rsq rLeng.w, res.w
	mul rLeng.w, rLeng.w, leng.w
	mul res.xyz, res.xyz, rLeng.w

// res.w has squared distance between s1 & s2
func AddNormalize3 res s1 s2
	locals rLeng
	add res.xyz, s1.xyz, s2.xyz
	Normalize3 res

func CalcCameraDir res
	AddNormalize3 res, c9, -v0

proc NHCalcer
	m4x4 oPos, v0, c10
	locals cameraDir, lightDir, halfVector
	CalcCameraDir cameraDir
	mad lightDir.xyz, -v0.xyz, c15.www, c15.xyz // r5 = (L),  world space 
	Normalize3 lightDir
	add halfVector.xyz, cameraDir.xyz, lightDir.xyz // r2 = not normalised half vector (H)
	mad oT0.xyz, v1, c3.xxx, c3.yyy // N
	mov oT1.xyz, halfVector  // H
	mad oT2.xyz, lightDir, c2.xxx, c2.xxx  // L

func CalcLocalVector res globalVec normal texU texV
	dp3 res.x, texU, globalVec
	dp3 res.y, texV, globalVec
	dp3 res.z, normal, globalVec          // r4 = local space normal

proc NHCalcBumped
	m4x4 oPos, v0, c10
	locals cameraDir, lightDir, halfVector, local, normal, texU, texV
	CalcVectors normal, texU, texV
	CalcCameraDir cameraDir
	mad lightDir.xyz, -v0.xyz, c15.www, c15.xyz // r5 = (L),  world space 
	Normalize3 lightDir
	add halfVector, lightDir, cameraDir // not normalised half vector (H)
	CalcLocalVector local halfVector normal texU texV
	mov oT1.xyz, local // H
	CalcLocalVector local lightDir normal texU texV
	Normalize3 local
	mad oT2.xyz, local, c2.xxx, c2.xxx  // L
	mul oT0.xy, v3, c6.xx // texture coords
	mov local.xy, c0
	mov local.z, c1

// store tex coords to T0&T1, dp3 & color params to D0 & D1
func CalcFastBump normal, texU, texV
	locals localLightDir
	CalcLocalVector localLightDir c15 normal texU texV
	Normalize3 localLightDir
	locals rTest
	dp3 rTest.x, normal, c15
	sge rTest.x, rTest.x, c0.x
	mul oD1.xyz, rTest.xxx, c14.xyz
	mad oD0.xyz, localLightDir.xyz, c2.xxx, c2.xxx
	mul oT0.xy, v3, c6.xx
	mul oT1.xy, v3, c6.xx

; params.x =  - distance to plane
; alpha = h / (R/4)
; params.w = fFall / ( 1 + h^2 )
; params.z = sqrt(1+h^2) - mapping w
func CalcPointLightMapping lightDir, params normal
	sub lightDir.xyz, c15, v0       // r0 = dir to light center
	dp3 params.x, lightDir, normal      // params.x = h(distance to plane)
	mul params.y, params.x, c21.z
	mul params.y, params.y, c4.z    // params.y = alpha
	mul params.y, params.y, params.y // alpha^2
	add params.w, params.y, c1.x
	rcp params.w, params.w
	rsq params.z, params.w
	mul params.y, params.y, params.y
	mad params.y, params.y, params.y, c4.w // alpha^8 + 4096
	rcp params.y, params.y
	mul params.y, params.y, c4.w
	mul params.w, params.y, params.w

// params.x = h - distance to plane

func CalcPointLight mulColor normal texU texV
	locals lightDir, temp
	CalcPointLightMapping lightDir, temp, normal
	sge lightDir.w, temp.x, c0.x       // params.z = h > 0
	mul lightDir.w, lightDir.w, temp.w   // params.z = final color
	mul oD0.xyz, lightDir.w, mulColor
	CalcPointLightTex oT0, lightDir, temp, texU, texV

func DoMirror outTex normal
	locals cameraDir, vFresnel
	CalcCameraDir cameraDir
	// r1 = cameraDir
	dp3 cameraDir.w, cameraDir, normal
	add vFresnel.w, c1.x, -cameraDir.w
	mul vFresnel.w, vFresnel.w, vFresnel.w
	mul vFresnel.w, vFresnel.w, vFresnel.w
	mul cameraDir.w, cameraDir.w, c2.z // cameraDir.w = 2 * ( cameraDir dp3 normal )
	mad oD0.w, vFresnel.w, c14.z, c14.w
	mad outTex.xyz, cameraDir.w, normal, -cameraDir.xyz
